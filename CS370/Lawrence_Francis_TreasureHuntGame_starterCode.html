def play_game(model, qmaze, pirate_cell, max_steps=None):
    qmaze.reset(pirate_cell)
    envstate = qmaze.observe()
    steps = 0
    if max_steps is None:
        max_steps = qmaze.maze.size * 4  # safety cutoff

    while steps < max_steps:
        state = np.asarray(envstate, dtype=np.float32)
        if state.ndim == 1:
            state = np.expand_dims(state, axis=0)

        q_values = model(state, training=False).numpy()
        action = np.argmax(q_values[0])

        envstate, reward, game_status = qmaze.act(action)
        steps += 1

        if game_status == 'win':
            return True
        elif game_status == 'lose':
            return False

    return False  # timed out with no result

def completion_check(model, maze_or_qmaze, max_steps=None):
    # Accept either raw numpy maze or TreasureMaze instance
    if isinstance(maze_or_qmaze, TreasureMaze):
        qmaze = maze_or_qmaze
    else:
        qmaze = TreasureMaze(maze_or_qmaze)

    for cell in qmaze.free_cells:
        if not qmaze.valid_actions(cell):
            continue
        if not play_game(model, qmaze, cell, max_steps=max_steps):
            return False
    return True

def build_model(maze):
    model = Sequential()
    model.add(Dense(maze.size, input_shape=(maze.size,)))
    model.add(PReLU())
    model.add(Dense(maze.size))
    model.add(PReLU())
    model.add(Dense(num_actions))
    model.compile(optimizer='adam', loss='mse')
    return model

@tf.function
def train_step(x, y):
    with tf.GradientTape() as tape:
        q_values = model(x, training=True)
        loss = loss_fn(y, q_values)
    grads = tape.gradient(loss, model.trainable_variables)
    optimizer.apply_gradients(zip(grads, model.trainable_variables))
    return loss

def format_time(seconds):
    if seconds < 400:
        return f"{seconds:.1f} seconds"
    elif seconds < 4000:
        m = seconds / 60.0
        return f"{m:.2f} minutes"
    else:
        h = seconds / 3600.0
        return f"{h:.2f} hours"

def qtrain(model, maze, **opt):
    # exploration factor
    global epsilon
   
    # Number of epochs
    n_epoch = opt.get('n_epoch', 15000)
   
    # Maximum memory to store episodes
    max_memory = opt.get('max_memory', 1000)
   
    # Maximum data size for training (batch size)
    data_size = opt.get('data_size', 50)
   
    # Frequency of target network updates (stabilizes training)
    target_update_freq = opt.get('target_update_freq', 50)
   
    # Start time for logging
    start_time = datetime.datetime.now()
   
    # Construct environment/game from numpy array: maze
    qmaze = TreasureMaze(maze)
   
    # Target Network to better guide training (double DQN style)
    target_model = clone_model(model)
    target_model.set_weights(model.get_weights())
   
    # Initialize experience replay object
    experience = GameExperience(model, target_model, max_memory=max_memory)
    
    win_history = []          # history of win/lose (1 or 0) per episode
    hsize = qmaze.maze.size // 2   # window size for recent win rate (~32)
    win_rate = 0.0
    
    loss = 0.0                # running loss for logging
    n_episodes = 0            # total episodes played
    
    # =============START_HERE================
    n_episodes = 0  # Counter for completed episodes (increment per episode, not per step)

    for epoch in range(n_epoch):
        # Random free cell start for better policy generalization across maze
        agent_cell = random.choice(qmaze.free_cells)
        qmaze.reset(agent_cell)
        
        # Initial state observation
        env_state = qmaze.observe()
        
        game_over = False  # Flag for episode termination
        
        while not game_over:
            previous_envstate = env_state
            
            # -greedy action selection: explore or exploit
            valid_acts = qmaze.valid_actions()  # Safe: only allowed moves (avoids penalties)
            if np.random.rand() < epsilon and valid_acts:
                action = random.choice(valid_acts)  # Exploration: uniform random valid action
            else:
                # Exploitation: greedy from main model prediction
                pred = experience.predict(previous_envstate)
                action = np.argmax(pred)
            
            # Step environment
            env_state, reward, game_status = qmaze.act(action)
            
            # Terminal state check
            game_over = game_status in ['win', 'lose']
            
            # Store transition for replay
            episode = [previous_envstate, action, reward, env_state, game_over]
            experience.remember(episode)
            
            # Train on minibatch if replay memory sufficient
            if len(experience.memory) >= data_size:
                inputs, targets = experience.get_data(data_size)
                batch_loss = train_step(inputs, targets)  # Custom TF gradient update
                loss = batch_loss.numpy()  # Update logged loss (last batch)
            
            # Periodic target network synchronization (hard copy for stability)
            if n_episodes > 0 and n_episodes % target_update_freq == 0:
                target_model.set_weights(model.get_weights())
        
        # Episode complete  increment counter
        n_episodes += 1
        
        # Log win for history
        win_history.append(1 if game_status == 'win' else 0)
        
        # Recent win rate (over fixed window or all early)
        recent_history = win_history[-hsize:] if len(win_history) > hsize else win_history
        win_rate = sum(recent_history) / len(recent_history) if recent_history else 0.0
        
        # Gradual epsilon decay every epoch (smooth exploration reduction)
        if epsilon > epsilon_min:
            epsilon *= epsilon_decay
        
        # Epoch progress print
        dt = datetime.datetime.now() - start_time
        t = format_time(dt.total_seconds())
        template = "Epoch: {:03d}/{:d} | Loss: {:.4f} | Episodes: {:d} | Win count: {:d} | Win rate: {:.3f} | time: {}"
        print(template.format(epoch, n_epoch - 1, loss, n_episodes, sum(win_history), win_rate, t))
        
        # Early stop on sustained high win rate + full maze validation
        if win_rate > 0.9 and completion_check(model, maze):
            print(f"Reached 100% win rate at epoch: {epoch}")
            break
    
    # Final training summary
    dt = datetime.datetime.now() - start_time
    seconds = dt.total_seconds()
    t = format_time(seconds)
    print("Training complete in:", t)
    
    # Return history, trained model, and env for testing
    return win_history, model, qmaze
    # =============END_HERE================
